/// USER
[-] getent group wheel | awk -F: '{ print $4 }' // list of admin user accounts seperate by comma
[-] getent passwd {1000..2000} | awk -F: '{ print $1 }' // list of all user accounts seperate by newline
[-] groups usrname // list of groups which contains this user
[-] chsh -l // list of available login shells
[-] gpasswd -a user wheel // change user to admin group
[-] gpasswd -d user wheel // remove user from admin group
[-] useradd -c fullname -N usrname // add new user
[] passwd usrname // set or change password for user
[-] usermod -u uid -g gname -l login_name -c fullname -s shell [-m -d home_dir] usrname // change user info
[-] userdel -r usrname // delete user and home dir

/// GROUP
[-] getent group {1000..2000} | awk -F: '{ print $1 }' // list of all user-defined group's name seperate by newline
[-] gpasswd -d user group // remove user from group
[-] gpasswd -a user group // add user to group
[-] gpasswd -M user1,user2,user3,... group // set list of group members
[-] gpasswd -M "$(getent group groupname | awk -F: '$4 {print $4","}')"user1,user2,user3,... groupname // add users to group
[-] groupadd groupname // add group with group name
[-] groupmod -n new_gname old_gname // change group name
[-] groupdel groupname // delete group
